<script>

var AREAS = ["fitness","money","career","mindset"];

var LABELS = {fitness:"Fitness",money:"Money",career:"Career",mindset:"Mindset"};

var ICONS  = {fitness:"o",money:"$",career:"^",mindset:"*"};

var COL = {

  fitness:{bg:"#d1fae5",text:"#065f46",bd:"#6ee7b7"},

  money:  {bg:"#dbeafe",text:"#1e3a8a",bd:"#93c5fd"},

  career: {bg:"#ede9fe",text:"#4c1d95",bd:"#c4b5fd"},

  mindset:{bg:"#fce7f3",text:"#831843",bd:"#f9a8d4"}

};


var goals = [];

var completed = {};

var evData = {reflection:"",wonder:"",tomorrow:""};

var allData = {};

var generating = false;

var API_KEY = “REPLACE_ME”;


function pad(n) { return n < 10 ? "0"+n : ""+n; }

function todayKey() {

  var d = new Date();

  return d.getFullYear()+"-"+pad(d.getMonth()+1)+"-"+pad(d.getDate());

}

function fmtDate(k) {

  return new Date(k+"T12:00:00").toLocaleDateString("en-US",{weekday:"long",month:"long",day:"numeric"});

}

function countDoneFor(c) {

  var n = 0;

  for (var i = 0; i < AREAS.length; i++) { if (c[AREAS[i]]) n++; }

  return n;

}

function loadStorage() {

  try { allData = JSON.parse(localStorage.getItem("bty_gh") || "{}"); } catch(e) { allData = {}; }

  var t = todayKey();

  if (allData[t]) {

    goals = allData[t].goals || [];

    completed = allData[t].completed || {};

    evData = allData[t].ev || {reflection:"",wonder:"",tomorrow:""};

  }

}

function saveStorage() {

  allData[todayKey()] = {goals:goals, completed:completed, ev:evData};

  localStorage.setItem("bty_gh", JSON.stringify(allData));

}

function go(id) {

  var all = document.querySelectorAll(".screen");

  for (var i = 0; i < all.length; i++) all[i].classList.remove("on");

  document.getElementById(id).classList.add("on");

  if (id === "home") renderHome();

  if (id === "history") renderHistory();

}

function getStreak() {

  var s = 0; var d = new Date();

  for (var i = 0; i < 60; i++) {

    var k = d.getFullYear()+"-"+pad(d.getMonth()+1)+"-"+pad(d.getDate());

    var e = allData[k];

    if (e && countDoneFor(e.completed||{}) > 0) { s++; d.setDate(d.getDate()-1); }

    else break;

  }

  return s;

}

function renderHome() {

  document.getElementById("h-date").textContent = fmtDate(todayKey());

  var s = getStreak();

  var sw = document.getElementById("h-streak");

  if (s > 0) {

    sw.innerHTML = '<div class="card" style="text-align:center;margin-bottom:20px"><div class="streak-n">'+s+'</div><div class="streak-l">day streak &mdash; keep going</div></div>';

  } else { sw.innerHTML = ""; }

  var mg = document.getElementById("h-mini");

  if (goals.length > 0) {

    var html = '<div class="mini-grid">';

    for (var i = 0; i < AREAS.length; i++) {

      var a = AREAS[i]; var done = completed[a]; var col = COL[a];

      var bg = done ? "background:"+col.bg+";border-color:"+col.bd+";" : "";

      var tc = done ? "color:"+col.text+";" : "";

      html += '<div class="mini" style="'+bg+'">';

      html += '<div class="mini-lbl" style="'+tc+'">'+ICONS[a]+" "+LABELS[a]+'</div>';

      html += '<div class="mini-val" style="'+tc+'">'+(done?"Done":"Pending")+'</div></div>';

    }

    html += '</div>';

    mg.innerHTML = html;

  } else { mg.innerHTML = ""; }

  document.getElementById("h-hist").style.display = Object.keys(allData).length > 1 ? "block" : "none";

}

function renderGoals() {

  var list = document.getElementById("g-list");

  var html = "";

  for (var i = 0; i < AREAS.length; i++) {

    var area = AREAS[i]; var col = COL[area]; var done = completed[area];

    var g = null;

    for (var j = 0; j < goals.length; j++) { if (goals[j].area === area) { g = goals[j]; break; } }

    if (!g) {

      html += '<div class="load-card"><span class="shimmer">'+(generating?"Generating "+LABELS[area].toLowerCase()+"...":"&mdash;")+'</span></div>';

    } else {

      var bg = done ? "background:"+col.bg+";border-color:"+col.bd+";" : "";

      var tc = done ? "color:"+col.text+";" : "";

      var cb = done ? "background:"+col.text+";border-color:"+col.text+";" : "";

      html += '<div class="gcard" style="'+bg+'" onclick="toggle(\''+area+'\')">';

      html += '<div class="hdr"><span class="tag" style="'+tc+'">'+LABELS[area]+'</span>';

      html += '<div class="circle" style="'+cb+'">'+(done?"&#10003;":"")+'</div></div>';

      html += '<div class="gtxt" style="'+tc+'">'+g.text+'</div></div>';

    }

  }

  list.innerHTML = html;

  var done2 = countDoneFor(completed);

  var total = goals.length || 4;

  var pct = Math.round((done2/total)*100);

  document.getElementById("pbar").style.width = pct+"%";

  document.getElementById("plbl").textContent = done2+" of "+total+" completed";

}

function toggle(area) {

  completed[area] = !completed[area];

  saveStorage();

  renderGoals();

}

function callAI(system, user, maxTokens, cb) {

  var xhr = new XMLHttpRequest();

  xhr.open("POST","https://api.anthropic.com/v1/messages",true);

  xhr.setRequestHeader("Content-Type","application/json");

  xhr.setRequestHeader("x-api-key", API_KEY);

  xhr.setRequestHeader("anthropic-version","2023-06-01");

  xhr.setRequestHeader("anthropic-dangerous-direct-browser-access","true");

  xhr.onreadystatechange = function() {

    if (xhr.readyState === 4) {

      try {

        var d = JSON.parse(xhr.responseText);

        cb(null, d.content && d.content[0] ? d.content[0].text : "");

      } catch(e) { cb(e,""); }

    }

  };

  xhr.onerror = function() { cb(new Error("Network error"),""); };

  xhr.send(JSON.stringify({

    model:"claude-haiku-4-5-20251001",

    max_tokens: maxTokens,

    system: system,

    messages:[{role:"user",content:user}]

  }));

}

function generate() {

  if (generating) return;

  generating = true;

  goals = []; completed = {}; evData = {reflection:"",wonder:"",tomorrow:""};

  go("goals");

  renderGoals();

  var usedGoals = [];

  var keys = Object.keys(allData).sort().reverse().slice(0,14);

  for (var i = 0; i < keys.length; i++) {

    var gs = allData[keys[i]].goals || [];

    for (var j = 0; j < gs.length; j++) usedGoals.push(gs[j].text);

  }

  var avoid = usedGoals.length > 0 ? "NEVER repeat these past goals: "+usedGoals.join(" | ") : "";

  var seed = ""+Date.now()+"_"+Math.floor(Math.random()*99999);

  var prompts = {

    fitness:"Give ONE creative fitness activity for a 22-year-old beginner today. Max 2 sentences. Think beyond basics.",

    money:"Give ONE money action a 22-year-old in debt can do today. Max 2 sentences. Be creative and specific.",

    career:"Give ONE career or job action for a 22-year-old with no job. Max 2 sentences. Be concrete.",

    mindset:"Give ONE mindset practice for a 22-year-old who feels behind. Max 2 sentences. Be gentle and specific."

  };

  var sys = "You are a direct creative life coach. Reply with ONE goal sentence only. No intro. Be specific and varied. Seed:"+seed+". "+avoid;

  var idx = 0;

  function next() {

    if (idx >= AREAS.length) { generating = false; saveStorage(); return; }

    var area = AREAS[idx];

    callAI(sys, prompts[area], 120, function(err, txt) {

      goals.push({area:area, text:txt||"Take one small step forward today."});

      renderGoals();

      idx++;

      next();

    });

  }

  next();

}

function goEvening() {

  go("evening");

  var done = countDoneFor(completed);

  document.getElementById("ev-sub").textContent = "You completed "+done+" of "+goals.length+" goals today";

  if (evData.reflection) { showEv(); return; }

  document.getElementById("ev-loading").style.display = "flex";

  var summary = "";

  for (var i = 0; i < goals.length; i++) {

    summary += LABELS[goals[i].area]+": "+goals[i].text+" -- "+(completed[goals[i].area]?"DONE":"skipped")+"\n";

  }

  var sys = "You are a wise warm mentor for a 22-year-old building their life. Reply ONLY with raw JSON no markdown. Format: {\"reflection\":\"3 honest warm sentences\",\"wonder\":\"2 poetic philosophical sentences about growth giving deep hope\",\"tomorrow\":\"2 sentences making them excited to wake up\"}";

  callAI(sys, "Today:\n"+summary, 400, function(err, txt) {

    try {

      var clean = txt.replace(/```json/g,"").replace(/```/g,"").trim();

      var p = JSON.parse(clean);

      evData = {

        reflection: p.reflection || "You showed up. That is everything.",

        wonder:     p.wonder     || "Every version of you that exists was built by a version that kept going.",

        tomorrow:   p.tomorrow   || "Tomorrow is a fresh page. You get to write it."

      };

    } catch(e) {

      evData = {

        reflection:"You showed up today. That matters more than you know.",

        wonder:"Every great life is simply a series of ordinary days where someone chose to keep going.",

        tomorrow:"Tomorrow you get another shot. Show up again."

      };

    }

    saveStorage();

    showEv();

  });

}

function showEv() {

  document.getElementById("ev-loading").style.display = "none";

  document.getElementById("ev-ref-txt").textContent = evData.reflection;

  document.getElementById("ev-ref").style.display = "block";

  document.getElementById("ev-won-txt").textContent = evData.wonder;

  document.getElementById("ev-won").style.display = "block";

  document.getElementById("ev-tom-txt").textContent = evData.tomorrow;

  document.getElementById("ev-tom").style.display = "block";

}

function renderHistory() {

  var days = Object.keys(allData).sort().reverse().slice(0,20);

  var html = "";

  for (var i = 0; i < days.length; i++) {

    var d = days[i]; var e = allData[d];

    var dn = countDoneFor(e.completed||{});

    var total = (e.goals||[]).length || 1;

    var pct = Math.round((dn/total)*100);

    var color = pct===100?"#065f46":pct>=50?"#4c1d95":"#9ca3af";

    html += '<div class="hi"><div><div class="hdate">'+fmtDate(d)+'</div><div class="hsub">'+dn+"/"+total+" goals completed</div></div>";

    html += '<div class="hpct" style="color:'+color+'">'+pct+'%</div></div>';

  }

  document.getElementById("hist-list").innerHTML = html || '<p style="color:#9ca3af;text-align:center">No history yet</p>';

}

document.getElementById("h-btn").onclick = generate;

document.getElementById("g-date").textContent = fmtDate(todayKey());

loadStorage();

renderHome();
